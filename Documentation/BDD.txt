// Creating tables

Table account {
  id int [pk, increment] // auto-increment
  login varchar
  password varchar //Don't forget to hash it
  type enum('administration', 'intervenant')
  first_name varchar
  name varchar
}

Table matiere {
  id int [pk, increment]
  name varchar
  specialite boolean
}

Table cours {
  id int [pk, increment]
  matiere_id int 
  _datetime datetime
}

Table affectation_matiere {
  id int [pk, increment]
  intervenant_id int
  //section_id int
  matiere_id int
  duree int
  intervalle_debut date
  intervalle_fin date 
}

Table to_exclude{
  id int [pk, increment]
  intervenant_id int
  matiere_id int //toutes les matieres = 0
  intervalle_debut date
  intervalle_fin date 
}
